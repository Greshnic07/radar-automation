import os import json import hashlib import time from datetime import datetime from playwright.sync_api import sync_playwright # --- НАСТРОЙКИ (БЕРУТСЯ ИЗ СЕКРЕТОВ GITHUB) --- LOGIN = os.environ.get("MY_LOGIN") PASSWORD = os.environ.get("MY_PASSWORD") # Если парсим Додо ИС, используем TOTP import pyotp TOTP_SECRET = os.environ.get("MY_TOTP") HASH_FILE = "reviews_hashes_combo.txt" DATA_FILE = "reviews.json" def get_hash(text): """Создает уникальный ID для текста отзыва""" return hashlib.md5(text.encode('utf-8')).hexdigest() def load_hashes(): if os.path.exists(HASH_FILE): with open(HASH_FILE, "r") as f: return set(line.strip() for line in f) return set() def save_new_hash(new_hash): with open(HASH_FILE, "a") as f: f.write(new_hash + "\n") def run(): print(f"[{datetime.now()}] Запуск сбора отзывов для Додо Пицца...") existing_hashes = load_hashes() new_reviews_found = [] with sync_playwright() as p: # headless=True важно для GitHub Actions browser = p.chromium.launch(headless=True) page = browser.new_page() try: # ТУТ ЛОГИКА ТВОЕГО ПАРСЕРА (Яндекс/2ГИС или Додо ИС) # Пример захода на страницу отзывов: page.goto("https://officemanager.dodois.io/Reports/UnitReviews") # Авторизация (если нужна) if "login" in page.url: page.fill("input[name='login']", LOGIN) page.fill("input[name='password']", PASSWORD) page.click("button[type='submit']") # Если есть 2FA: if TOTP_SECRET: totp = pyotp.TOTP(TOTP_SECRET) page.fill("input[name='code']", totp.now()) page.keyboard.press("Enter") time.sleep(5) # Ждем прогрузки # СОБИРАЕМ ОТЗЫВЫ (пример селектора) review_elements = page.locator(".review-card").all() for el in review_elements: text = el.inner_text() h = get_hash(text) if h not in existing_hashes: print(f"Нашли новый отзыв: {h}") new_reviews_found.append({ "id": h, "text": text, "date": datetime.now().strftime("%d.%m.%Y %H:%M") }) save_new_hash(h) if new_reviews_found: # Загружаем старые отзывы из json и добавляем новые в начало all_data = [] if os.path.exists(DATA_FILE): with open(DATA_FILE, "r", encoding="utf-8") as f: all_data = json.load(f) updated_data = new_reviews_found + all_data # Оставляем только последние 20 отзывов, чтобы файл не раздувался with open(DATA_FILE, "w", encoding="utf-8") as f: json.dump(updated_data[:20], f, ensure_ascii=False, indent=4) print(f"Добавлено {len(new_reviews_found)} новых отзывов.") else: print("Новых отзывов не обнаружено.") except Exception as e: print(f"Ошибка при парсинге: {e}") finally: browser.close() if __name__ == "__main__": run()
